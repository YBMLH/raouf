Live sharing — one-time setup

Work on the same sheet with someone else, live, from any phone. Needs a free Supabase account (~5 min, email or GitHub login only — no credit card).

  1. Go to supabase.com → Start your project → sign in → New project (any name, set a database password, pick a region) → wait ~2 min for it to provision.
  2. Open the SQL Editor (left sidebar) → New query → paste the SQL below → Run:
create table if not exists public.workspaces (
  code text primary key,
  json text not null,
  client text,
  updated_at timestamptz default now()
);

alter table public.workspaces enable row level security;

create policy "allow all" on public.workspaces
  for all using (true) with check (true);

alter publication supabase_realtime add table public.workspaces;
